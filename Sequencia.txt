int numero <- 5.
int a <- 0.
int b <- 1.
int soma <- a + b.
int contador <- 3.

while (contador <= numero) {
    int temp <- a + b.
    a <- b.
    b <- temp.
    soma <- soma + b.
    contador <- contador + 1.
}.
